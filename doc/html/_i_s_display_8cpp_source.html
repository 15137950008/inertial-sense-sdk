107 void cInertialSenseDisplay::ProcessData(p_data_t *data, bool enableReplay, double replaySpeedX)
108 {
109  int curTimeMs = current_weekMs();
110 
111  if (enableReplay)
112  {
113  static bool isTowMode = false;
114  static int gpsTowMsOffset = 0;
115  static int msgTimeMsOffset = 0;
116  int msgTimeMs = 0;
117 
118  // Copy only new data
119  uDatasets d = {};
120  copyDataPToStructP(&d, data, sizeof(uDatasets));
121 
122  // Record message time. In either ToW or time since boot.
123  switch (data->hdr.id)
124  {
125  // Time of week - double
126  case DID_INS_MISC:
127  case DID_INS_1:
128  case DID_INS_2:
129  msgTimeMs = (int)(1000.0*d.ins1.timeOfWeek);
130  isTowMode = true;
131  break;
132 
133  // Time of week - uint32 ms
134  case DID_SYS_PARAMS:
135  case DID_GPS_POS:
136  msgTimeMs = d.gps.pos.timeOfWeekMs;
137  isTowMode = true;
138  break;
139 
140  case DID_GPS:
141  msgTimeMs = d.gps.pos.timeOfWeekMs;
142  gpsTowMsOffset = (int)(1000.0*d.gps.towOffset);
143  isTowMode = true;
144  break;
145 
146  // Time since boot - double
147  case DID_MAGNETOMETER_1:
148  case DID_BAROMETER:
149  case DID_SYS_SENSORS:
150  case DID_DELTA_THETA_VEL:
151  case DID_DUAL_IMU:
152  case DID_EKF_STATES:
153  case DID_IMU_2:
154  case DID_IMU_1:
155  if( isTowMode )
156  msgTimeMs = (int)(1000.0*d.imu.time) + gpsTowMsOffset;
157  else
158  msgTimeMs = (int)(1000.0*d.imu.time);
159  break;
160 
161  // Unidentified data type
162 // default: printf("Unknown DID %d\t", data->hdr.id); return;
163  }
164 
165 
166  // Control replay rate
167  if (msgTimeMs != 0 && replaySpeedX>0.0)
168  {
169  for (;;)
170  {
171  curTimeMs = current_weekMs();
172 
173  // Replay speed
174  int replayTimeMs = (int)(long)(((double)curTimeMs)*replaySpeedX);
175 
176  // Reinitialize message offset
177  if (abs(msgTimeMs + msgTimeMsOffset - replayTimeMs) > 1500)
178  msgTimeMsOffset = replayTimeMs - msgTimeMs;
179 
180  // Proceed if we're caught up
181  if (replayTimeMs >= msgTimeMs + msgTimeMsOffset)
182  break;
183 
184  // Add delay
185 // SLEEP_US(1000);
186  SLEEP_MS(10);
187  }
188  }
189  }
190 
191 
192  static int timeSinceRefreshMs = 0;
193  static int timeSinceClearMs = 0;
194  static char idHist[DID_COUNT] = { 0 };
195  if (m_displayMode != DMODE_SCROLL)
196  {
197  // Clear display every 2 seconds or if we start seeing new messages.
198  if (curTimeMs - timeSinceClearMs > 2000 || curTimeMs < timeSinceClearMs || idHist[data->hdr.id] == 0)
199  {
200  Clear();
201  idHist[data->hdr.id] = 1;
202  timeSinceClearMs = curTimeMs;
203  }
204  }
205 
206 
207  // Display Data
208 
209 
210  switch (m_displayMode)
211  {
212  default:
213  m_displayMode = DMODE_PRETTY;
214  case DMODE_PRETTY:
215 
216  // Data stays at fixed location (no scroll history)
217  DataToVector(data);
218 
219  // Limit printData display updates to 20Hz (50 ms)
220  if (curTimeMs - timeSinceRefreshMs > 50 || curTimeMs < timeSinceRefreshMs)
221  {
222  Home();
223  if (enableReplay)
224  cout << Replay(replaySpeedX) << endl;
225  else
226  cout << Connected() << endl;
227 
228  cout << VectortoString();
229 
230  timeSinceRefreshMs = curTimeMs;
231  }
232  break;
233 
234  case DMODE_STATS:
235  // Limit printData display updates to 20Hz (50 ms)
236  if (curTimeMs - timeSinceRefreshMs > 50 || curTimeMs < timeSinceRefreshMs)
237  {
238  Home();
239  cout << Connected() << endl;
240  DataToStats(data);
241 
242  timeSinceRefreshMs = curTimeMs;
243  }
244  break;
245 
246 
247  case DMODE_SCROLL: // Scroll display
248  cout << DataToString(data) << endl;
249  break;
250  }
251

285 void cInertialSenseDisplay::DataToStats(const p_data_t* data)
286 {
287  size_t id = data->hdr.id;
288  if (m_didStats.size() <= id)
289  { // Resize vector if necessary
290  m_didStats.resize(id + 1);
291  }
292 
293  // Update stats
294  int curTimeMs = current_weekMs();
295  sDidStats &s = m_didStats[id];
296  s.count++;
297  if(s.lastTimeMs)
298  s.dtMs = curTimeMs - s.lastTimeMs;
299  s.lastTimeMs = curTimeMs;
300 
301  // Display stats
302  printf("Data ID Count dt\n");
303  for (int i = 0; i < (int)m_didStats.size(); i++)
304  {
305  sDidStats &s = m_didStats[i];
306  if (s.count)
307  {
308  printf(" %6d %8d %10.3lf\n", i, s.count, s.dtMs*0.001);
309  }
310  }

314 string cInertialSenseDisplay::DataToString(const p_data_t* data)
315 {
316  uDatasets d = {};
317 
318  // Copy only new data
319  copyDataPToStructP(&d, data, sizeof(uDatasets));
320 
321  string str;
322  switch (data->hdr.id)
323  {
324  case DID_DEV_INFO: str = DataToStringDevInfo(d.devInfo, data->hdr); break;
325  case DID_DUAL_IMU: str = DataToStringDualIMU(d.dualImu, data->hdr); break;
326  case DID_IMU_1:
327  case DID_IMU_2: str = DataToStringIMU(d.imu, data->hdr); break;
328  case DID_DELTA_THETA_VEL: str = DataToStringDThetaVel(d.dThetaVel, data->hdr); break;
329  case DID_INS_1: str = DataToStringINS1(d.ins1, data->hdr); break;
330  case DID_INS_2: str = DataToStringINS2(d.ins2, data->hdr); break;
331  case DID_MAGNETOMETER_1:
332  case DID_MAGNETOMETER_2: str = DataToStringMag(d.mag, data->hdr); break;
333  case DID_BAROMETER: str = DataToStringBaro(d.baro, data->hdr); break;
334  case DID_GPS: str = DataToStringGPS(d.gps, data->hdr); break;
335  case DID_SYS_PARAMS: str = DataToStringSysParams(d.sysParams, data->hdr); break;
336  case DID_SYS_SENSORS: str = DataToStringSysSensors(d.sysSensors, data->hdr); break;
337 
338  default:
339  char buf[128];
340  SNPRINTF(buf, 128, "DID: %d - No Print Setup\n", data->hdr.id);
341  str = buf;
342  break;
343  }
344 
345  return str;
346 }

349 string cInertialSenseDisplay::DataToStringINS1(const ins_1_t &ins1, const p_data_hdr_t& hdr)
350 {
351  (void)hdr;
352  char buf[BUF_SIZE];
353  int n = SNPRINTF(buf, BUF_SIZE, "DID_INS_1: %.3lfs %dwk", ins1.timeOfWeek, ins1.week);
354 
355  if (m_displayMode == DMODE_SCROLL)
356  { // Single line format
357  n += SNPRINTF(&buf[n], BUF_SIZE - n, ", %s theta[%6.2f,%6.2f,%7.2f], uvw[%6.2f,%6.2f,%6.2f], lla[%12.7f,%12.7f,%7.1f], ned[%6.3f,%6.3f,%6.3f]",
358  ((ins1.iStatus&INS_STATUS_ATT_ALIGNED) ? "aligned" : "_______"),
359  ins1.theta[0] * C_RAD2DEG_F,
360  ins1.theta[1] * C_RAD2DEG_F,
361  ins1.theta[2] * C_RAD2DEG_F,
362  ins1.uvw[0], ins1.uvw[1], ins1.uvw[2],
363  ins1.lla[0], ins1.lla[1], ins1.lla[2],
364  ins1.ned[0], ins1.ned[1], ins1.ned[2]);
365  }
366  else
367  { // Spacious format
368  n += SNPRINTF(&buf[n], BUF_SIZE - n, "\n\tEuler\t");
369  n += SNPRINTF(&buf[n], BUF_SIZE - n, PRINTV3_P2,
370  ins1.theta[0] * C_RAD2DEG_F, // Roll
371  ins1.theta[1] * C_RAD2DEG_F, // Pitch
372  ins1.theta[2] * C_RAD2DEG_F); // Yaw
373  n += SNPRINTF(&buf[n], BUF_SIZE - n, "\tUWV\t");
374  n += SNPRINTF(&buf[n], BUF_SIZE - n, PRINTV3_P1,
375  ins1.uvw[0], // U body velocity
376  ins1.uvw[1], // V body velocity
377  ins1.uvw[2]); // W body velocity
378  n += SNPRINTF(&buf[n], BUF_SIZE - n, "\tLLA\t");
379  n += SNPRINTF(&buf[n], BUF_SIZE - n, PRINTV3_LLA,
380  ins1.lla[0], // INS Latitude
381  ins1.lla[1], // INS Longitude
382  ins1.lla[2]); // INS Ellipsoid altitude (meters)
383  // SNPRINTF(&buf[n],BUF_SIZE-n, "\tiStatus 0x%08x\n", g_msg.ins1.iStatus);
384  n += SNPRINTF(&buf[n], BUF_SIZE - n, "\tAligned Att %d Vel %d Pos %d\n",
385  (ins1.iStatus & INS_STATUS_ATT_ALIGNED) != 0,
386  (ins1.iStatus & INS_STATUS_VEL_ALIGNED) != 0,
387  (ins1.iStatus & INS_STATUS_POS_ALIGNED) != 0
388  );
389  }
390 
391  return buf;
