38 // [COMM INSTRUCTION] 4.) This function is called every time there is new data.
39 void cltool_dataCallback(InertialSense* i, p_data_t* data)
40 {
41  // Print data to terminal
42  g_inertialSenseDisplay.ProcessData(data);
43 
44  // uDatasets is a union of all datasets that we can receive. See data_sets.h for a full list of all available datasets.
45  uDatasets d = {};
46  copyDataPToStructP(&d, data, sizeof(uDatasets));
47 
48  // Example of how to access dataset fields.
49  switch (data->hdr.id)
50  {
51  case DID_INS_2:
52  d.ins2.qn2b; // quaternion attitude
53  d.ins2.uvw; // body velocities
54  d.ins2.lla; // latitude, longitude, altitude
55  break;
56  case DID_INS_1:
57  d.ins1.theta; // euler attitude
58  d.ins1.lla; // latitude, longitude, altitude
59  break;
60  case DID_DUAL_IMU: d.dualImu; break;
61  case DID_DELTA_THETA_VEL: d.dThetaVel; break;
62  case DID_IMU_1: d.imu; break;
63  case DID_IMU_2: d.imu; break;
64  case DID_GPS: d.gps; break;
65  case DID_MAGNETOMETER_1: d.mag; break;
66  case DID_MAGNETOMETER_2: d.mag; break;
67  case DID_BAROMETER: d.baro; break;
68  case DID_SYS_SENSORS: d.sysSensors; break;
69  }
70 }

72 // Where we tell the uINS what data to send and at what rate.
73 // "cltool_dataCallback()" is registered as the callback functions for all received data.
74 // All DID messages are found in data_sets.h
75 void cltool_setupCommunications(InertialSense& inertialSenseInterface)
76 {
77  int periodMs = 50;
78  inertialSenseInterface.StopBroadcasts(); // Stop streaming any prior messages
79 
80  // ask for device info every 2 seconds
81  inertialSenseInterface.BroadcastBinaryData(DID_DEV_INFO, 2000, cltool_dataCallback);
82 
83  // depending on command line options. stream various data sets
84  if (g_commandLineOptions.streamSol)
85  {
86  inertialSenseInterface.SetBroadcastSolutionEnabled(true);
87  }
88  if (g_commandLineOptions.streamINS1)
89  {
90  inertialSenseInterface.BroadcastBinaryData(DID_INS_1, periodMs, cltool_dataCallback);
91  }
92  if (g_commandLineOptions.streamINS2)
93  {
94  inertialSenseInterface.BroadcastBinaryData(DID_INS_2, periodMs, cltool_dataCallback);
95  }
96  if (g_commandLineOptions.streamSysSensors)
97  {
98  inertialSenseInterface.BroadcastBinaryData(DID_SYS_SENSORS, 100, cltool_dataCallback);
99  }
100  if (g_commandLineOptions.streamDualIMU)
101  {
102  inertialSenseInterface.BroadcastBinaryData(DID_DUAL_IMU, periodMs, cltool_dataCallback);
103  }
104  if (g_commandLineOptions.streamIMU1)
105  {
106  inertialSenseInterface.BroadcastBinaryData(DID_IMU_1, periodMs, cltool_dataCallback);
107  }
108  if (g_commandLineOptions.streamIMU2)
109  {
110  inertialSenseInterface.BroadcastBinaryData(DID_IMU_2, periodMs, cltool_dataCallback);
111  }
112  if (g_commandLineOptions.streamDThetaVel)
113  {
114  inertialSenseInterface.BroadcastBinaryData(DID_DELTA_THETA_VEL, periodMs, cltool_dataCallback);
115  }
116  if (g_commandLineOptions.streamGPS)
117  {
118  inertialSenseInterface.BroadcastBinaryData(DID_GPS, 200, cltool_dataCallback);
119  }
120  if (g_commandLineOptions.streamMag1)
121  {
122  inertialSenseInterface.BroadcastBinaryData(DID_MAGNETOMETER_1, periodMs, cltool_dataCallback);
123  }
124  if (g_commandLineOptions.streamBaro)
125  {
126  inertialSenseInterface.BroadcastBinaryData(DID_BAROMETER, periodMs, cltool_dataCallback);
127  }
128  if (g_commandLineOptions.serverHostAndPort.size() != 0)
129  {
130  inertialSenseInterface.OpenServerConnectionRTCM3(g_commandLineOptions.serverHostAndPort);
131  }

135 int inertialSenseMain()
136 {
137  g_inertialSenseDisplay.SetDisplayMode(g_commandLineOptions.displayMode);
138 
139  // if replay data log specified on command line, do that now and return
140  if (g_commandLineOptions.replayDataLog)
141  {
142  return !cltool_replayDataLog();
143  }
144 
145  // [COMM INSTRUCTION] 1.) Create InertialSense object and open serial port.
146  InertialSense inertialSenseInterface(cltool_dataCallback);
147  if (!inertialSenseInterface.Open(g_commandLineOptions.comPort.c_str(), g_commandLineOptions.baudRate))
148  {
149  cout << "Failed to open serial port at " << g_commandLineOptions.comPort.c_str() << endl;
150  return -1; // Failed to open serial port
151  }
152 
153  // if bootloader was specified on the command line, do that now and return out
154  if (g_commandLineOptions.bootloaderFileName.length() != 0)
155  {
156  // [BOOTLOADER INSTRUCTIONS] Update firmware
157  char bootloaderError[1024];
158  cout << "Bootloading file at " << g_commandLineOptions.bootloaderFileName << endl;
159  bool success = inertialSenseInterface.BootloadFile(g_commandLineOptions.bootloaderFileName, bootloadUploadProgress, bootloadVerifyProgress, bootloaderError, sizeof(bootloaderError));
160  if (!success)
161  {
162  cout << "Error bootloading file " << g_commandLineOptions.bootloaderFileName << ", error: " << bootloaderError;
163  }
164  return (success ? 0 : -1);
165  }
166 
167  // [COMM INSTRUCTION] 2.) Enable data broadcasting from uINS
168  cltool_setupCommunications(inertialSenseInterface);
169 
170  // [LOGGER INSTRUCTION] Setup and start data logger
171  cltool_setupLogger(inertialSenseInterface);
172 
173  // catch ctrl-c, break, etc.
174  cltool_setupCtrlCHandler();
175 
176  // clear display
177  g_inertialSenseDisplay.Clear();
178 
179  try
180  {
181  // Main loop. Could be in separate thread if desired.
182  while (!g_ctrlCPressed)
183  {
184  // [COMM INSTRUCTION] 3.) Process data and messages
185  inertialSenseInterface.Update();
186 
187  // Specify the minimum time between read/write updates.
188  SLEEP_MS(1);
189  }
190  }
191  catch (...)
192  {
193  cout << "Unknown exception, ";
194  }
195 
196  cout << "Shutting down..." << endl;
197 
198  // close the interface cleanly, this ensures serial port and any logging are shutdown properly
199  inertialSenseInterface.Close();
200 
201  return 0;
202 }
