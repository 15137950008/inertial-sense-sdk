45  pfnHandleBinaryData handler = s->binaryCallback[data->hdr.id];

60  if (data->hdr.id == DID_DEV_INFO)
61  {
62  memcpy(s->devInfo, data->buf, sizeof(dev_info_t));
63  }
64  }

113 void InertialSense::LoggerThread()
114 {
115  bool running = true;
116  while (running)
117  {
118  SLEEP_MS(20);
119 
120  // gather up packets in memory
121  vector<p_data_t> packets;
122  {
123  // lock so we can read and clear m_logPackets
124  lock_guard<mutex> l(m_logMutex);
125  for (list<p_data_t>::iterator i = m_logPackets.begin(); i != m_logPackets.end(); i++)
126  {
127  packets.push_back(*i);
128  }
129 
130  // clear shared memory
131  m_logPackets.clear();
132 
133  // update running state
134  running = m_logger.Enabled();
135  }
136 
137  // log the packets
138  for (size_t i = 0; i < packets.size(); i++)
139  {
140  m_logger.LogData(0, &packets[i].hdr, packets[i].buf);
141  }
142  }
143 }

145 void InertialSense::StepLogger(InertialSense* i, const p_data_t* data)
146 {
147  // single threaded implementation would just do this:
148  // i->m_logger.LogData(0, (p_data_hdr_t*)&data->hdr, (uint8_t*)data->buf); return;
149  const time_t reinitSeconds = 2;
150  {
151  lock_guard<mutex> l(i->m_logMutex);
152  if (i->m_logger.Enabled())
153  {
154  i->m_logPackets.push_back(*data);
155  }
156  }
157 
158  // if time has passed, re-send the appropriate log solution message
159  if (time(0) - i->m_lastLogReInit > reinitSeconds)
160  {
161  sendDataComManagerInstance(i->m_cmHandle, 0, DID_CONFIG, &i->m_logSolution, sizeof(i->m_logSolution), offsetof(config_t, sLogCtrl));
162  i->m_lastLogReInit = time(0);
163  }
164 }

166 bool InertialSense::Open(const char* port, int baudRate)
167 {
168  // clear the device in preparation for auto-baud
169  memset(&m_deviceInfo, 0, sizeof(m_deviceInfo));
170 
171  if (validateBaudRate(baudRate) != 0 || serialPortOpen(&m_serialPort, port, baudRate, 0) != 1)
172  {
173  return false;
174  }
175 
176  // negotiate baud rate by querying device info - don't return out until it negotiates or times out
177  // if the baud rate is already correct, the request for the message should succeed very quickly
178  time_t startTime = time(0);
179 
180  // try to auto-baud for up to 3 seconds, then abort if we didn't get a valid packet
181  // we wait until we get a valid serial number and manufacturer
182  while (m_deviceInfo.serialNumber == 0 && m_deviceInfo.manufacturer[0] == '\0' && time(0) - startTime < 3)
183  {
184  getDataComManagerInstance(m_cmHandle, 0, DID_DEV_INFO, 0, 0, 0);
185  SLEEP_MS(2); // 2 milliseconds wait for reply
186  stepComManagerInstance(m_cmHandle);
187  }
188 
189  if (m_deviceInfo.serialNumber == 0 && m_deviceInfo.manufacturer[0] == '\0')
190  {
191  Close();
192  return false;
193  }
194 
195  return true;
196 }

221 bool InertialSense::OpenServerConnectionRTCM3(const string& hostAndPort)
222 {
223  size_t colon = hostAndPort.find(':', 0);
224  if (colon == string::npos)
225  {
226  return false;
227  }
228  size_t colon2 = hostAndPort.find(':', colon + 1);
229  if (colon2 == string::npos)
230  {
231  return false;
232  }
233  string host = hostAndPort.substr(0, colon);
234  string portString = hostAndPort.substr(colon + 1, colon2 - colon - 1);
235  int port = (int)strtol(portString.c_str(), NULL, 10);
236  bool opened = (m_tcpClient.Open(host, port) == 0);
237  if (opened)
238  {
239  // try to find url:user:pwd on the string
240  colon = hostAndPort.find(':', colon2 + 1);
241  if (colon != string::npos)
242  {
243  string url = hostAndPort.substr(colon2 + 1, colon - colon2 - 1);
244  colon2 = hostAndPort.find(':', colon + 1);
245  if (colon2 != string::npos)
246  {
247  string user = hostAndPort.substr(colon + 1, colon2 - colon - 1);
248  string pwd = hostAndPort.substr(colon2 + 1);
249  if (user.size() != 0 && pwd.size() != 0)
250  {
251  m_tcpClient.HttpGet(url, "InertialSense", user, pwd);
252  }
253  }
254  }
255  }
256  return opened;
257 }
